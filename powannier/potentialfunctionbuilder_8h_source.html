134  _potential._coefficients.clear();
135  Position xmin(_potential.dim());
136  Position xmax(_potential.dim());
137  xmin.fill(-0.5);

140  // calculate Fourier series coefficients (V_n) for a given potential,
141  // that is V(r) = \sum_n V_n e^{i (\sum_i n_i \vec{b_i}) \vec{r}},
142  // where n is a set of indices (summation is performed for all
143  // integer indices in the range [-cutoff, cutoff]),
144  // and \vec{b_i} are reciprocal lattice basis vectors
145  for (NPoint n : nspace(_cutoff, _potential.dim())) {
146  arma::cx_double coefficient = chop(IntegrationProvider::integrate(
147  [&, n] (Position alpha) {
148  Position r = alpha * _potential.latticeBasis();
149  return _potentialFunction(r) * 1.0 *
150  std::exp(arma::cx_double(0, -2 * pi * arma::dot(alpha, n)));
151  }, xmin, xmax, _precision, _precision));
152 
153  if (coefficient != 0.0) {
154  _potential._coefficients.push_back(std::make_pair(n, coefficient));
155  }
156  }
157 
158  if (_maxError >= 0) {
159  if (relativeError<IntegrationProvider>() > _maxError) {
160  throw std::runtime_error("Relative error of approximated potential exceeded the requirements.");
161  }
162  }
163 
164  return std::move(PotentialBuilder::complete());
165  }
