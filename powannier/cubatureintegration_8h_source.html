86  static bool checkIfIntegrationFirstStepOk(Function&& function, Vector xmin, Vector xmax, int dim, double prec_rel, double prec_abs, int level) {
87  if (level > 5) {
88  return true;
89  }

92  xmid[i] = 0.5 * (xmin[i] + xmax[i]);

103  // lambda_i are random numbers in [0.4, 0.6] range

105  static auto splitIntegrate(Function&& function, Vector xmin, Vector xmax, int dim, double prec_rel, double prec_abs, int level) {
106  std::random_device r;
107  std::default_random_engine e1(r());
108  std::uniform_real_distribution<double> uniform_dist(0.4, 0.6);
109 
110  typename std::result_of<Function(Vector)>::type integral = 0;
111 
112  Vector xmid = xmin;
113  for (int i = 0; i < dim; ++i) {
114  xmid[i] = xmin[i] + uniform_dist(e1) * (xmax[i] - xmin[i]);
115  }
116 
117  for (int i = 0; i < pow(2, dim); ++i) {
118  int ih = i;
119  auto xminh = xmin;
120  auto xmaxh = xmax;
121  for (int j = 0; j < dim; ++j) {
122  if (ih % 2 == 0) {
123  xmaxh[j] = xmid[j];
124  } else {
125  xminh[j] = xmid[j];
126  }
127  ih /= 2;
128  }
129  integral += integrate(std::ref(function), xminh, xmaxh, prec_rel, prec_abs, level+1);
130  }
131 
132  return integral;
133  }
